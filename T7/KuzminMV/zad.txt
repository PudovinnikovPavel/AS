struct number                                                                         
{
  int value;
  number *next;
  number *prev;
};

void del(number **p1, number **p2, number *pc)              
{
  number *pp, *nn;

  if(pc==NULL) return;
  if((pc!=*p1) && (pc!=*p2))
  {
   pp=pc->prev;
   nn=pc->next;
   pp->next=nn;
   nn->prev=pp;
  }
   else if(pc==*p1)
   {
   *p1=(*p1)->next;
   (*p1)->prev=NULL;                                                           
   }
    else if(pc==*p2)
    {
     *p2=(*p2)->prev;
     (*p2)->next=NULL;                                                         
    }
  free(pc);

}

void printlist(number *p1)                                                     
{
  while (p1!=NULL)
  {
    cout << p1->value << " ";
    p1 = p1->next;
  }
  cout << '\n';
}
