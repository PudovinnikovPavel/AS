#include <iostream.h>
#include <stdlib.h>

void my_sort(int *m, int n)
{
	int key = 0, j = 0;
	for (int i = 1; i < n; i++)
	{
		key = m[i];
		j = i - 1;
		while (j >= 0 && m[j] > key)
		{
			m[j + 1] = m[j];
			j--;
		}
		m[j + 1] = key;
	}
	return;      }
